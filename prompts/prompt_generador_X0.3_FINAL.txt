(function(){
  const EXEC_SUMMARY = `[TEXTO COMPLETO DEL BLOQUE A]`;
  const popup = document.getElementById('execPopup');
  const popupTx = document.getElementById('execPopupText');
  let hideTimer = null;
  function showExec(){ if(hideTimer){clearTimeout(hideTimer);hideTimer=null;} popupTx.textContent=EXEC_SUMMARY; popup.style.display='block'; }
  function scheduleHide(){ hideTimer=setTimeout(function(){popup.style.display='none';},300); }
  function cancelHide(){ if(hideTimer){clearTimeout(hideTimer);hideTimer=null;} }
  ['triggerLogo','triggerNombre','triggerFrase'].forEach(function(id){
    const el=document.getElementById(id); if(!el)return;
    el.style.cursor='pointer';
    el.addEventListener('mouseenter',showExec);
    el.addEventListener('mouseleave',scheduleHide);
  });
  popup.addEventListener('mouseenter',cancelHide);
  popup.addEventListener('mouseleave',scheduleHide);
  document.addEventListener('click',function(e){ if(!popup.contains(e.target)) popup.style.display='none'; });
})();
</script>

- FCF es NON-GAAP. Mostrar como KPI PARALELO. NUNCA como flujo lineal del P&L.
- Discrepancia >2% entre fuentes → usar dato del 10-K.
- NUNCA inventar splits que la empresa no reporta.
